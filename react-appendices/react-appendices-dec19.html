Appendix 1: Key Developer Concepts

  • map()
    const myArray = [1,2,3,4]
// Increase value of myArray of each by 1
myArray.map(el=>el+1)
// Four "b"'s
myArray.map(()=>"b")
  • Promises
    const myPromise = new Promise((resolve,reject)) => {
	if (false) {
		set Timeout(() => {
			resolve('I have succeeded');
		}, 1000); 
	} else {
		reject('I have failed');
	}
});

myPromise
	.then(value => console.log(value))
	.then(newValue => newValue + '!!!!')
	.then(nv => console.log(nv))
	.catch(rejectValue => console.log(rejectValue))
    • Example of Promise
      fetch('https://jsonplaceholder.typicode.com/todos/1')

  • filter()
    const myArray = [1,3,5,7,9]
// Filter out elements in array less than 5
myArray.filter(el => el > 4)
// Returns [5, 7, 9]
myArray.filter(el => False)
// Doesn't return anything
  • includes()
    • Primitive Types
      • Strings
      • Booleans
      • Null
      • Undefined
      • Number
      • Symbol
        • Example
          var b = 3
var c = b

        • Example (comparing spots in memory)
          const obj1 = { id: 1 }
const obj2 = { id: 1 }
obj1 == obj2
// false
const obj3 = obj2
obj3 == obj2
// true
    const myArray = [1,2,3,4,5]
myArray.includes(3)
// True
myArray.includes(9)
// False
// Could also include index
myArray.includes(2,3) // Does the element "2" exist at our 3rd index?
// False
myArray.includes(2,1)
// True
const newArray = [{id:1}, {id:2}, {id:3}]
newArray.includes({id:2})
// False
